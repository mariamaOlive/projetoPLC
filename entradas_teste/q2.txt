//Funcionalidades do programa:
//Funções recursivas: fibonacci e fatorial

function fat(n) {

	if(n==0) {
		return 1;
	} else {
		return n*fat(n-1);
	}
}

function fib(n) {
	if(n <= 2) {
		return 1;
	} else {
		return fib(n-1) + fib(n-2);
	}
}
